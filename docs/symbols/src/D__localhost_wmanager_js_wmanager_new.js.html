514 		// http://kevin.vanzonneveld.net
515 		// +   original by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
516 		// +   improved by: FremyCompany
517 		// +   improved by: Onno Marsman
518 		// *     example 1: isset( undefined, true);

520 		// *     example 2: isset( 'Kevin van Zonneveld' );
521 		// *     returns 2: true
522 
523 		var a=arguments, l=a.length, i=0;
524 
525 		if (l===0) {
526 			throw new Error('Empty isset');
527 		}
528 
529 		while (i!==l) {
530 			if (typeof(a[i])=='undefined' || a[i]===null) {
531 				return false;
532 			} else {
533 				i++;
534 			}
535 		}
